```js
import { RawTable, RawTh, RawTd, RawTr, RawThead, RawTbody } from '@strapi/parts/RawTable';
```

    {() => {
      const ROW_COUNT = 30;
      const COL_COUNT = 5;
      const rows = Array(ROW_COUNT - 1)

            .map((e, x) => x),

      return (

                </RawTh>

                </RawTh>
                <RawTh>
                  <Box padding={2} background="neutral200">
                    Four
                  </Box>
                </RawTh>
                <RawTh>
                  <Box padding={2} background="neutral200">
                    Five
                  </Box>
                </RawTh>
              </RawTr>
            </RawThead>
            <RawTbody>
              {rows.map((row, rowIndex) => (

              ))}
            </RawTbody>
          </RawTable>
        </Box>
      );
    }}
