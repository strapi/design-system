# BaseRadio

The `BaseRadio` component is the basis of the `Radio` component.

## Usage

<Canvas>
  <Story name="base">
    {() => {
      const {
        colors: { neutral800 },
      } = useTheme();
      const [selected, setSelected] = useState();
      return (
        <div>
          <Typography variant="beta" textColor="neutral800" id="trophy-champions">
            Make a choice
          </Typography>
          <RadioGroup
            labelledBy="trophy-champions"
            onChange={(e) => setSelected(e.target.value)}
            value={selected}
            name="prefered-meal"
          >
            <div>
              <label style={{ color: neutral800 }} htmlFor="pizza">
                Pizza
              </label>
              <BaseRadio value="pizza" id="pizza" />
            </div>
            <div>
              <label style={{ color: neutral800 }} htmlFor="bagel">
                Bagel
              </label>
              <BaseRadio value="bagel" id="bagel" />
            </div>
          </RadioGroup>
        </div>
      );
    }}
  </Story>
</Canvas>
